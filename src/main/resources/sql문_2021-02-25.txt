CREATE TABLE product(
    num int PRIMARY KEY not null,
    name VARCHAR(20) NOT NULL,
    maker VARCHAR(20) NOT NULL,

    origin VARCHAR(20) NOT NULL,

    category2 bigint REFERENCES category2(id) on delete cascade,
